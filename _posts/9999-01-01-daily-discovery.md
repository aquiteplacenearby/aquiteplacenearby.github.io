```python
# unwrap decorator
def plus_one(func):
    def wrapper(arg, kwarg):
        return func(arg, kwarg) + 1        
    wrapper.unwrapper = func
    return wrapper

@plus_one
def func(x):
    return x
    
# usage example
func.unwrapper(1) # 1
func(1) # 2
```
